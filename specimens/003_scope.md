# Specimen 003: Scope #

    function one() {
      for (var i = 1; i <= 10; ++i) {
        console.log(i);
      }
    }

    function two() {
      for (var i = 1; i <= 10; ++i) {
        (function() {
          console.log(i);
        })();
      }
    }

    function three() {
      for (var i = 1; i <= 10; ++i) {
        setTimeout(function() {
          console.log(i);
        }, 0);
      }
    }
